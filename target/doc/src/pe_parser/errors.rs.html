1use core::fmt;
2
3/// Custom error type for Parsey.
4#[derive(Debug)]
5pub enum Error {
6    /// DOS header magic is wrong (expected `MZ` / 0x5A4D).
7    InvalidMagic(u16),
8
9    /// PE signature didn’t match `PE\0\0` (0x00004550).
10    InvalidPeSignature(u32),
11
12    /// Optional header magic wasn’t 0x10B (PE32) or 0x20B (PE32+).
13    UnsupportedOptionalHeader(u16),
14
15    /// An underlying I/O operation failed.
16    Io(std::io::Error),
17
18    /// A table’s RVA/offset fell outside the file data.
19    InvalidTableOffset,
20
21    /// Failed to read a null-terminated DLL name.
22    InvalidDllName,
23
24    /// Data in the buffer didn’t make sense.
25    InvalidData,
26
27    /// Buffer was smaller than we expected.
28    InvalidSize { expected: usize, found: usize },
29}
30
31impl fmt::Display for Error {
32    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
33        match self {
34            Error::InvalidMagic(m) =>
35                write!(f, "Bad DOS magic: 0x{m:04X}"),
36            Error::InvalidPeSignature(s) =>
37                write!(f, "Bad PE signature: 0x{s:08X}"),
38            Error::UnsupportedOptionalHeader(m) =>
39                write!(f, "Unknown optional-header magic: 0x{m:04X}"),
40            Error::Io(e) =>
41                write!(f, "I/O error: {e}"),
42            Error::InvalidTableOffset =>
43                write!(f, "Table offset is out of bounds"),
44            Error::InvalidDllName =>
45                write!(f, "Couldn’t read DLL name"),
46            Error::InvalidData =>
47                write!(f, "Malformed data in buffer"),
48            Error::InvalidSize { expected, found } =>
49                write!(f, "Expected at least {expected} bytes, found {found}"),
50        }
51    }
52}
53
54impl core::error::Error for Error {}
55
56impl From<std::io::Error> for Error {
57    fn from(err: std::io::Error) -> Self {
58        Error::Io(err)
59    }
60}
61
62/// Result type for PE parser operations.
63pub type Result<T> = core::result::Result<T, Error>;
