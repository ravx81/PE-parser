1use crate::headers::{FileHeader, SectionHeader};
2use std::ptr;
3use crate::errors::Result;
4impl SectionHeader{
5     /// Parse all section headers declared in the COFF FileHeader.
6    ///
7    /// * `buffer` – complete PE image in memory.
8    /// * `file_header` – already parsed COFF FileHeader containing
9    ///   `number_of_sections` and `size_of_optional_header`.
10    /// * `e_lfanew` – offset from DOS header to PE signature.
11    ///
12    /// Calculates the start of the section table as:
13    /// `e_lfanew + 4 + size_of::<FileHeader>() + file_header.size_of_optional_header as usize`.
14    /// Then allocates a Vec and reads each `SectionHeader` via `ptr::read_unaligned`.
15    ///
16    /// # Safety
17    ///
18    /// Uses unaligned, direct memory reads. Caller must ensure:
19    /// - `buffer.len() >= section_offset + number_of_sections * size_of::<SectionHeader>()`
20    /// - The bytes in that range are valid `SectionHeader` entries.
21    pub fn parse_section_headers(buffer: &[u8], file_header: &FileHeader, e_lfanew: usize, ) -> Result<Vec<SectionHeader>>{
22        //section_offset = e_lfanew + PE_signature... 
23        let section_offset = e_lfanew + 4 + size_of::<crate::headers::FileHeader>() + file_header.size_of_optional_header as usize;
24        let number_of_sections = file_header.number_of_sections as usize;
25        let mut sections = Vec::with_capacity(number_of_sections);
26    
27        // first section_header stars here
28        let base_ptr = unsafe { buffer.as_ptr().add(section_offset) as *const SectionHeader};
29    
30        for i in 0..number_of_sections{
31            let ptr_section = unsafe {base_ptr.add(i)};
32    
33            let section = unsafe {ptr::read_unaligned(ptr_section)};
34    
35            sections.push(section);
36        }
37        Ok(sections)
38    }
39}
